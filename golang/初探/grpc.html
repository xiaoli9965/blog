        # 小试牛刀 Protocol Buffer

        syntax = "proto3";

package alex;
option go_package = "alex/xxx";

import "google/protobuf/timestamp.proto";

message Person {
    string name = 1;
    int32 id = 2;  // Unique ID number for this person.
    string email = 3;

    enum PhoneType {
        MOBILE = 0;
        HOME = 1;
        WORK = 2;
    }

    message PhoneNumber {
        string number = 1;
        PhoneType type = 2;
    }

    repeated PhoneNumber phones = 4;

    google.protobuf.Timestamp last_updated = 5;
}

// Our address book file is just one of these.
message AddressBook {
    repeated Person people = 1;
}

        安装依赖

        go get github.com/golang/protobuf/protoc-gen-go 
go get google.golang.org/grpc/cmd/protoc-gen-go-grpc

        :::: code-group ::: code-group-item 编写 proto

        syntax = "proto3";
package alexrpc;
option go_package = "/user";

message UserReq {
    int32 user_id = 1;// 用户id
}
message UserInfo {
    int64 uid = 1;
    string account_id = 2;
    string user_name = 3;
}

service UsersService {
    rpc GetUserInfo(UserReq) returns (UserInfo);
}

        ::: ::: code-group-item 生成代码

        protoc --go_out=plugins=grpc:./alex-grpc/ ./alex-grpc/pd/*.proto

        :::

        ::: code-group-item client

        package main

import (
	"alex-rpc/alex-grpc/user"
	"context"
	"fmt"
	"google.golang.org/grpc"
)

func main() {
	conn, err := grpc.Dial(":8989", grpc.WithInsecure())
	if err != nil {
		fmt.Println("failed to grpc1: ", err)
		return
	}
	defer conn.Close()

	ct := user.NewUsersServiceClient(conn)
	stock, err := ct.GetUserInfo(context.Background(), &user.UserReq{UserId: 12})
	if err != nil {
		fmt.Println("failed to grpc2: ", err)
	}
	fmt.Println(stock)
}

        ::: ::: code-group-item server

        package main

import (
	"alex-rpc/alex-grpc/user"
	"context"
	"fmt"
	"google.golang.org/grpc"
	"net"
)

type UserService struct {
}

func (_ *UserService) GetUserInfo(ctx context.Context, in *user.UserReq) (*user.UserInfo, error) {
	fmt.Println(ctx)
	return &user.UserInfo{Uid: 10, AccountId: "usr-xxx", UserName: "alex"}, nil
}
func main() {
	rc := grpc.NewServer()
	user.RegisterUsersServiceServer(rc, new(UserService))

	lis, err := net.Listen("tcp", "127.0.0.1:8989")
	if err != nil {
		fmt.Println("failed to listen: ", err)
	}
	defer lis.Close()
	fmt.Println("ok")
	rc.Serve(lis)
}
